FUNCTION void getInput(char name[], char number[], char &symbol)
    DEFINE a boolean validity = false

    WHILE (!validity)
        DISPLAY "Input a name you can easily remember: "
        INPUT name[]
        IF (!isalpha(name[])) THEN
            DISPLAY "Error!! Expected alphabets only! Try again."
        ELSE
            DISPLAY "Enter a four digit number: "
            INPUT number[]
            IF (cin.fail() OR strlen(number) != 4) THEN
                cin.clear()
                cin.ignore(numeric_limits<streamsize::max(), '\n')
                DISPLAY "Error! Uknown numbers! Try again."
            ELSE
                DISPLAY "Enter a keyboard accessible symbol: "
                INPUT symbol
                IF (cin.fail())
                    cin.clear()
                    cin.ignore(numeric_limits<streamsize>::max(), '\n')
                    DISPLAY "Error!! You have not entered a symbol. Please try again."
                ELSE
                    SET validity = true
                END IF
            END IF
        END IF
    END WHILE
END FUNCTION
